# Lanieats Merchant App — Claude code Build Prompt

---

## PROJECT OVERVIEW

Build the **Lanieats Merchant App** — a React Native mobile application for restaurant owners and food vendors to manage their restaurants, menus, and incoming orders on the Lanieats food delivery platform in Nigeria.

This is a **production-ready** app. Build everything fully functional, fully connected to the API, and with a polished, professional UI.

## API SETUP

### Base URL
```
http://localhost:8000/api/v1
```

### `src/api/client.js`
```js
import axios from 'axios';
import { getToken, clearAuth } from '../utils/storage';

const BASE_URL = 'http://localhost:8000/api/v1';

const client = axios.create({
  baseURL: BASE_URL,
  timeout: 15000,
  headers: { 'Content-Type': 'application/json' },
});

client.interceptors.request.use(async (config) => {
  const token = await getToken();
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

client.interceptors.response.use(
  (res) => res,
  async (error) => {
    if (error.response?.status === 401) {
      await clearAuth();
      // Trigger navigation to login from your auth store
    }
    return Promise.reject(error);
  }
);

export default client;
```

### `src/utils/storage.js`
```js
import AsyncStorage from '@react-native-async-storage/async-storage';

export const saveAuth = async (token, user) => {
  await AsyncStorage.multiSet([
    ['token', token],
    ['user', JSON.stringify(user)],
  ]);
};

export const getToken = async () => AsyncStorage.getItem('token');

export const getUser = async () => {
  const user = await AsyncStorage.getItem('user');
  return user ? JSON.parse(user) : null;
};

export const clearAuth = async () => {
  await AsyncStorage.multiRemove(['token', 'user']);
};
```

## SOCKET.IO SETUP

### `src/services/socket.js`
```js
import { io } from 'socket.io-client';

const SOCKET_URL = 'http://localhost:8000';
let socket = null;

export const connectSocket = () => {
  if (!socket) {
    socket = io(SOCKET_URL, {
      transports: ['websocket'],
      reconnection: true,
      reconnectionAttempts: 5,
      reconnectionDelay: 1000,
    });

    socket.on('connect', () => console.log('🔌 Socket connected'));
    socket.on('disconnect', () => console.log('❌ Socket disconnected'));
  }
  return socket;
};

export const getSocket = () => socket;

export const joinMerchantRooms = (restaurantIds) => {
  const s = getSocket();
  if (s) {
    restaurantIds.forEach((id) => {
      s.emit('join_merchant_room', id);
    });
  }
};

export const disconnectSocket = () => {
  if (socket) {
    socket.disconnect();
    socket = null;
  }
};
```

## ZUSTAND STORES

### `src/store/auth.store.js`
```js
import { create } from 'zustand';
import { saveAuth, clearAuth, getToken, getUser } from '../utils/storage';
import { loginApi, registerApi, getMeApi } from '../api/auth.api';
import { connectSocket, joinMerchantRooms, disconnectSocket } from '../services/socket';

const useAuthStore = create((set, get) => ({
  user: null,
  token: null,
  isAuthenticated: false,
  isLoading: true,

  initialize: async () => {
    try {
      const token = await getToken();
      const user = await getUser();
      if (token && user) {
        set({ user, token, isAuthenticated: true });
        connectSocket();
      }
    } catch (e) {
      console.log('Init error', e);
    } finally {
      set({ isLoading: false });
    }
  },

  login: async (email, password) => {
    const res = await loginApi(email, password);
    const { token, user } = res.data.data;
    await saveAuth(token, user);
    set({ user, token, isAuthenticated: true });
    connectSocket();
  },

  register: async (data) => {
    const res = await registerApi({ ...data, role: 'MERCHANT' });
    const { token, user } = res.data.data;
    await saveAuth(token, user);
    set({ user, token, isAuthenticated: true });
    connectSocket();
  },

  logout: async () => {
    await clearAuth();
    disconnectSocket();
    set({ user: null, token: null, isAuthenticated: false });
  },
}));

export default useAuthStore;
```

### `src/store/order.store.js`
```js
import { create } from 'zustand';
import { getRestaurantOrdersApi, updateOrderStatusApi } from '../api/order.api';

const useOrderStore = create((set, get) => ({
  orders: [],
  activeFilter: 'ALL',
  isLoading: false,

  fetchOrders: async (restaurantId, status = null) => {
    set({ isLoading: true });
    try {
      const res = await getRestaurantOrdersApi(restaurantId, status);
      set({ orders: res.data.data.orders });
    } finally {
      set({ isLoading: false });
    }
  },

  setFilter: (filter) => set({ activeFilter: filter }),

  addOrder: (order) => {
    set((state) => ({ orders: [order, ...state.orders] }));
  },

  updateOrderStatus: async (orderId, status) => {
    await updateOrderStatusApi(orderId, status);
    set((state) => ({
      orders: state.orders.map((o) =>
        o.id === orderId ? { ...o, status } : o
      ),
    }));
  },

  removeOrder: (orderId) => {
    set((state) => ({
      orders: state.orders.filter((o) => o.id !== orderId),
    }));
  },
}));

export default useOrderStore;
```

```js
// Format Nigerian currency
export const formatCurrency = (amount) => {
  return `₦${Number(amount).toLocaleString('en-NG')}`;
};

// Format order number from UUID
export const formatOrderNumber = (id) => {
  return `#${id.slice(-4).toUpperCase()}`;
};

// Format relative time
export const formatRelativeTime = (date) => {
  const seconds = Math.floor((new Date() - new Date(date)) / 1000);
  if (seconds < 60) return 'Just now';
  if (seconds < 3600) return `${Math.floor(seconds / 60)} mins ago`;
  if (seconds < 86400) return `${Math.floor(seconds / 3600)}h ago`;
  return new Date(date).toLocaleDateString('en-NG', { day: 'numeric', month: 'short' });
};

// Format time (24h to 12h)
export const formatTime = (time) => {
  if (!time) return '';
  const [hours, minutes] = time.split(':');
  const h = parseInt(hours);
  const ampm = h >= 12 ? 'PM' : 'AM';
  return `${h % 12 || 12}:${minutes} ${ampm}`;
};
```

## ERROR HANDLING PATTERN

Apply this pattern to every API call throughout the app:

```js
try {
  setLoading(true);
  const res = await api.call();
  // handle success
} catch (error) {
  const message = error.response?.data?.message || 'Something went wrong';
  
  // Show validation errors per field if present
  if (error.response?.data?.errors) {
    error.response.data.errors.forEach((err) => {
      setError(err.field, { message: err.message });
    });
    return;
  }

  // Show toast for general errors
  showToast({ type: 'error', message });
} finally {
  setLoading(false);
}
```

## BUSINESS RULES TO ENFORCE IN UI

1. **Merchant must be approved** — if `merchant.isApproved` is false, show a "Pending Approval" banner on the restaurant screen. They can still manage menus but cannot receive orders.

2. **Order status flow is one-directional** — once an order is PREPARING, do not show a button to go back to CONFIRMED.

3. **Only PENDING and CONFIRMED orders can be cancelled** by the merchant — hide cancel button for any other status.

4. **Merchant status mapping for PATCH /orders/:id/status:**
   - From `CONFIRMED` or `PENDING` → next action is `PREPARING`
   - From `PREPARING` → next action is `READY_FOR_PICKUP`
   - Note: `CONFIRMED` is now set automatically by payment — merchant's first manual action is `PREPARING`

5. **isOpen toggle** — make it very obvious when the restaurant is closed. Show a full-width amber banner on the Orders screen: "Your restaurant is currently closed. Customers cannot place orders."

6. **Real-time orders tab badge** — count PENDING + CONFIRMED orders for the badge number on the Orders tab.
